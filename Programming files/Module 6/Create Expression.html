Create Expression

        Functions for creating expressions are provided in the ExpressionCollection class, as follows:
    
        
     When creating expressions of a specified type, the type is indicated by a string, which can be one of “Number”,
    “String”, “Boolean”, “Integer”, “Point” or “Vector”. The following code shows how these functions are used:

Example Code


        using NXOpen;

        namespace Create_Expression

        {
            public class CreateExpression
            {
                public static void Main(string[] args)
                {
                    Session theSession = Session.GetSession();
                    Part workPart = theSession.Parts.Work;
        
                    NXOpen.Unit mmUnit = workPart.UnitCollection.GetBase("Length");
                    NXOpen.Unit cmUnit = workPart.UnitCollection.FindObject("CentiMeter");
        
                   //Get the ExpressionCollection of the work part 
        
                    NXOpen.ExpressionCollection ExpCollection = workPart.Expressions;
        
        
                    //Create Three expressions 
                    Expression exp1 = ExpCollection.Create("X=12");
                    Expression exp2 = ExpCollection.CreateExpression("Integer", "Length=12");
                    Expression exp3 = ExpCollection.CreateWithUnits("Width=13", cmUnit);
        
                }
        
                public static int GetUnloadOption(string args)
                {
                    return (int)NXOpen.Session.LibraryUnloadOption.Immediately;
                }
            }
        }
