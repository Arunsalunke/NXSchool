Sphere Builder


            using NXOpen;
            namespace SphereBuilder
            {
                public class SphereBuilder
                {
                    public static void Main(string[] args)
                    {
                        Session theSession = Session.GetSession();
                        Part workPart = theSession.Parts.Work;
            
                        NXOpen.Features.Sphere newSphere = null;
                        NXOpen.Features.SphereBuilder builder = workPart.Features.CreateSphereBuilder(newSphere);
                        builder.Type = NXOpen.Features.SphereBuilder.Types.CenterPointAndDiameter;
                        Point CenterPoint = workPart.Points.CreatePoint(new Point3d(0, 0, 0));
                        builder.Diameter.Value = 100;
                        builder.BooleanOption.Type = NXOpen.GeometricUtilities.BooleanOperation.BooleanType.Create;
                        NXOpen.NXObject sphere = builder.Commit();
                        builder.Destroy();
                    }
                    
                    public static int GetUnloadOption(string args)
                    {
                        return 1;
                    }
                }
            }
