Cylinder Builder


            using NXOpen;

            namespace CylinderBuilder

            {
                public class CylindricalBuilder
                {
                    public static void Main(string[] args)
                    {
                        Session theSession = Session.GetSession();
                        Part workPart = theSession.Parts.Work;
            
                        NXOpen.Features.Cylinder newCylinder = null;
                        NXOpen.Features.CylinderBuilder builder;
                        builder=workPart.Features.CreateCylinderBuilder(newCylinder);
                        builder.Type = NXOpen.Features.CylinderBuilder.Types.AxisDiameterAndHeight;
            
                        builder.Direction = new Vector3d(1, 0, 0);
                        builder.Origin = new Point3d(0, 0, 0);
                        builder.Diameter.Value = 100;
                        builder.Height.Value = 50;
                        builder.BooleanOption.Type = NXOpen.GeometricUtilities.BooleanOperation.BooleanType.Create;
            
                        NXOpen.NXObject Cylinder = builder.Commit();
                        builder.Destroy();
                    }
            
                    public static int GetUnloadOption(string args)
                    {
                        return 1;
                    }

                }

            }
